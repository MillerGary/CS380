	pfH 		INTEGER NOT NULL,
	pfN 		VARCHAR(75) NOT NULL,
	cH 		INTEGER NOT NULL,
	cN 		VARCHAR(75) NOT NULL,
	playoff 	VARCHAR(4) NOT NULL
);
